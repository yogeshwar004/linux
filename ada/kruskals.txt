#include<stdio.h>
void kruskals(int cost[][10],int n)
{
	int i,j,ec=0,par[n],min,u,v,a,b,mincost=0;
	for(i=0;i<n;i++)
		par[i]=-1;
	while(ec<n-1)
	{
		min=999;
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
               		{
				if(cost[i][j]<min)
				{
					min=cost[i][j];
					a=u=i;
					b=v=j;
				}
	               	}
		}
		while(par[u]!=-1)
			u=par[u];
		while(par[v]!=-1)
			v=par[v];
		if(u!=v)
		{
			printf("\nEdge from vertex %d to %d is %d",a,b,min);
			par[b]=a;
			ec++;
			mincost+=min;
		}
		cost[a][b]=cost[b][a]=999;
	}
	printf("\nMincost=%d\n",mincost);
}

{
	int n,i,j,cost[10][10];

	scanf("%d",&n);

		for(j=0;j<n;j++)
			scanf("%d",&cost[i][j]);
